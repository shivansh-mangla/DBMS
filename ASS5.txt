CREATE TABLE DEPT(
    DEPTNO NUMBER(3) PRIMARY KEY,
    DEPT_NAME VARCHAR(20),
    CONSTRAINT DEPT_NAME_CONSTRAINT CHECK (DEPT_NAME IN ('ACC', 'COMP', 'ELECT'))
);


CREATE TABLE EMP(
    EMPNO NUMBER(4) PRIMARY KEY,
    EMP_NAME VARCHAR(20) UNIQUE,
    JOB VARCHAR(20),
    SAL INT NOT NULL,
    DEPTNO NUMBER(3),
    MGR_NO,
    CONSTRAINT JOB_CONSTRAINT CHECK (JOB IN  ('PROF', 'AP', 'LECT')),
    CONSTRAINT DEPTNO_FRN_KEY FOREIGN KEY(DEPTNO) REFERENCES DEPT(DEPTNO),
    CONSTRAINT MGR_NO_FRN_KEY FOREIGN KEY(MGR_NO) REFERENCES EMP(EMPNO)
);

ALTER TABLE EMP
    DROP CONSTRAINT JOB_CONSTRAINT;
